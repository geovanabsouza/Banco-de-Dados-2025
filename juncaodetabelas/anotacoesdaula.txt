ANOTAÇÕES JUNÇÃO DE TABELAS:

CREATE DATABASE estacionamento;
DROP DATABASE estacionamneto;

CREATE TABLE cliente (
    codcliente int AUTO_INCREMENT,
    nome varchar(100) not null,
    datanascimento date not null,
    cpf varchar(20) not null,
    rg varchar(20) not null,
    telefone varchar(20) not null,
    email varchar (100) not null);
    
CREATE TABLE modelo (
    codmodelo int AUTO_INCREMENT,
    modelo varchar (30) not null);
    
CREATE TABLE andar (
    codlugar int AUTO_INCREMENT,
    capacidade varchar(100) not null, 
    andar varchar(20) not null);
    
CREATE TABLE veiculo (
    codplaca int AUTO_INCREMENT,
    cor varchar(20) not null,
    FOREIGN KEY (cpf) REFERENCES cliente (cpf),
    FOREIGN KEY (codmodelo) REFERENCES modelo (codmodelo));
    
CREATE estaciona (
    codestaciona int AUTO_INCREMENT,
    horaentrada datetime not null,
    horasaida datetime not null,
    dataentrada date not null,
    datasaida date not null,
    FOREIGN KEY (codplaca) REFERENCES veiculo(codplaca),
    FOREIGN KEY (codlugar) REFERENCES andar (codlugar));
